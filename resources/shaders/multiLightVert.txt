#version 410
in vec3 a_Position;
in vec3 a_Normal;
in vec2 a_TexCoord;

out vec3 v_FragPos;
out vec3 v_Normal;
out vec2 v_TexCoord;
												 
uniform mat4 projection;													 
uniform mat4 model;														 
uniform mat4 view;									 												 
																					 
void main()																 
{																			 
	v_FragPos = vec3(model * vec4(a_Position, 1.0));						 
	v_Normal = mat3(transpose(inverse(model))) * a_Normal;
	v_TexCoord = a_TexCoord;
	
	gl_Position = projection * view * vec4(v_FragPos, 1.0);
}
