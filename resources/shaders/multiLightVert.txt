#version 410

in vec3 a_Position;
in vec2 a_TexCoord;
in vec3 a_Normal;	

out vec3 v_FragPos;													 
out vec3 v_Normal;		
out vec2 v_TexCoord;

uniform mat4 model;														 
uniform mat4 view;			
uniform mat4 projection;
																					 
void main()																 
{																			 
	v_FragPos = vec3(model * vec4(a_Position, 1.0));						 
	v_TexCoord = a_TexCoord;	 
	v_Normal = mat3(transpose(inverse(model))) * a_Normal;
	
	gl_Position = projection * view * model * vec4(a_Position, 1.0);
}
