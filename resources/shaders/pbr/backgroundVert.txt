#version 410

layout (location = 0) in vec3 a_Position;
uniform mat4 projection;
uniform mat4 view;

out vec3 v_WorldPos;

void main()
{
	v_WorldPos = a_Position;

	mat4 rotView = mat4(mat3(view));
	vec4 clipPos = projection * rotView * vec4(v_WorldPos, 1.0);

	gl_Position = clipPos.xyww;
}
